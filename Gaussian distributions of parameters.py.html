import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

# Mean values
mu_theta = 0
mu_x = 0.8
mu_hd = 1.6

# Standard deviations from the table
sigma_theta_expert = 0.0001
sigma_x_expert = 0.001
sigma_hd_expert = 0.001

sigma_theta_novice = 0.001
sigma_x_novice = 0.01
sigma_hd_novice = 0.01

# Create a range of values for each variable
theta_range = np.linspace(mu_theta - 3*sigma_theta_novice, mu_theta + 3*sigma_theta_novice, 1000)
x_range = np.linspace(mu_x - 3*sigma_x_novice, mu_x + 3*sigma_x_novice, 1000)
hd_range = np.linspace(mu_hd - 3*sigma_hd_novice, mu_hd + 3*sigma_hd_novice, 1000)

# Calculate the Gaussian distributions
theta_dist_expert = norm.pdf(theta_range, mu_theta, sigma_theta_expert)
x_dist_expert = norm.pdf(x_range, mu_x, sigma_x_expert)
hd_dist_expert = norm.pdf(hd_range, mu_hd, sigma_hd_expert)

theta_dist_novice = norm.pdf(theta_range, mu_theta, sigma_theta_novice)
x_dist_novice = norm.pdf(x_range, mu_x, sigma_x_novice)
hd_dist_novice = norm.pdf(hd_range, mu_hd, sigma_hd_novice)

# Set y-axis limits
y_limit = [0, 4000]

# Plot for Î¸ (theta)
plt.figure(figsize=(7, 5))
plt.plot(theta_range, theta_dist_expert, label='Expert', color='blue')
plt.plot(theta_range, theta_dist_novice, label='Novice', color='red', linestyle='--')
plt.xlabel(r'$\theta$ (rad)', fontsize=25)
plt.ylabel('Probability Density', fontsize=25)
plt.tick_params(axis='both', which='major', labelsize=20)
plt.ylim(y_limit)
plt.legend(fontsize=20)
plt.grid(True)
plt.show()

# Plot for x (extension)
plt.figure(figsize=(7, 5))
plt.plot(x_range, x_dist_expert, label='Expert', color='blue')
plt.plot(x_range, x_dist_novice, label='Novice', color='red', linestyle='--')
plt.xlabel(r'$x$ (m)', fontsize=25)
plt.ylabel('Probability Density', fontsize=25)
plt.tick_params(axis='both', which='major', labelsize=20)
plt.ylim(y_limit)
plt.legend(fontsize=20)
plt.grid(True)
plt.show()

# Plot for h_d (draw height)
plt.figure(figsize=(7, 5))
plt.plot(hd_range, hd_dist_expert, label='Expert', color='blue')
plt.plot(hd_range, hd_dist_novice, label='Novice', color='red', linestyle='--')
plt.xlabel(r'$h_d$ (m)', fontsize=25)
plt.ylabel('Probability Density', fontsize=25)
plt.tick_params(axis='both', which='major', labelsize=20)
plt.ylim(y_limit)
plt.legend(fontsize=20)
plt.grid(True)
plt.show()
